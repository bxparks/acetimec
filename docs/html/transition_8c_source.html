99 {
100  const int16_t kOneDayAsMinutes = 60 * 24;
101 
102  if (dt->minutes <= -kOneDayAsMinutes) {
103  AtcLocalDate ld = { dt->year, dt->month, dt->day };
104  atc_local_date_decrement_one_day(&ld);
105  dt->year = ld.year;
106  dt->month = ld.month;
107  dt->day = ld.day;
108  dt->minutes += kOneDayAsMinutes;
109  } else if (kOneDayAsMinutes <= dt->minutes) {

112  dt->year = ld.year;

114  dt->day = ld.day;
115  dt->minutes -= kOneDayAsMinutes;
116  } else {
117  // do nothing

121 uint8_t atc_date_tuple_compare_fuzzy(
122  const AtcDateTuple *t,
123  const AtcDateTuple *start,
124  const AtcDateTuple *until)
125 {
126  // Use int32_t because a delta year of 2730 or greater will exceed
127  // the range of an int16_t.
128  int32_t t_months = t->year * (int32_t) 12 + t->month;
129  int32_t start_months = start->year * (int32_t) 12 + start->month;
130  if (t_months < start_months - 1) return kAtcMatchStatusPrior;
131  int32_t until_months = until->year * 12 + until->month;
132  if (until_months + 1 < t_months) return kAtcMatchStatusFarFuture;
133  return kAtcMatchStatusWithinMatch;
134 }

309 const char *atc_transition_extract_letter(const AtcTransition *t)
310 {
311  // RULES column is '-' or hh:mm, so return NULL to indicate this.
312  if (t->rule == NULL) {
313  return NULL;
314  }
315 
316  // RULES point to a named rule, and LETTER is a single, printable character.
317  // Return the letter_buf which contains a NUL-terminated string containing the
318  // single character, as initialized in
319  // atc_processing_create_transition_for_year().
320  char letter = t->rule->letter;
321  if (letter >= 32) {
322  return t->letter_buf;
323  }
324 
325  // RULES points to a named rule, and the LETTER is a string. The
326  // rule->letter is a non-printable number < 32, which is an index into
327  // a list of strings given by match->era->zonePolicy->letters[].
328  const AtcZonePolicy *policy = t->match->era->zone_policy;
329  uint8_t num_letters = policy->num_letters;
330  if (letter >= num_letters) {
331  // This should never happen unless there is a programming error. If it
332  // does, return an empty string. (createTransitionForYear() sets
333  // letterBuf to a NUL terminated empty string if rule->letter < 32)
334  return t->letter_buf;
335  }
336 
337  // Return the string at index 'rule->letter'.
338  return policy->letters[(uint8_t) letter];
339 }

343 uint8_t atc_transition_compare_to_match(
344  const AtcTransition *t, const AtcMatchingEra *match)
345 {
346  // Find the previous Match offsets.
347  int16_t prev_match_offset_minutes;
348  int16_t prev_match_delta_minutes;
349  if (match->prev_match) {
350  prev_match_offset_minutes = match->prev_match->last_offset_minutes;
351  prev_match_delta_minutes = match->prev_match->last_delta_minutes;
352  } else {
353  prev_match_offset_minutes = atc_zone_era_std_offset_minutes(match->era);
354  prev_match_delta_minutes = 0;
355  }
356 
357  // Expand start times.
358  AtcDateTuple stw;
359  AtcDateTuple sts;
360  AtcDateTuple stu;
361  atc_date_tuple_expand(
362  &match->start_dt,
363  prev_match_offset_minutes,
364  prev_match_delta_minutes,
365  &stw,
366  &sts,
367  &stu);
368 
369  // Transition times.
370  const AtcDateTuple *ttw = &t->transition_time;
371  const AtcDateTuple *tts = &t->transition_time_s;
372  const AtcDateTuple *ttu = &t->transition_time_u;
373 
374  // Compare Transition to Match, where equality is assumed if *any* of the
375  // 'w', 's', or 'u' versions of the DateTuple are equal. This prevents
376  // duplicate Transition instances from being created in a few cases.
377  if (atc_date_tuple_compare(ttw, &stw) == 0
378  || atc_date_tuple_compare(tts, &sts) == 0
379  || atc_date_tuple_compare(ttu, &stu) == 0) {
380  return kAtcMatchStatusExactMatch;
381  }
382 
383  if (atc_date_tuple_compare(ttu, &stu) < 0) {
384  return kAtcMatchStatusPrior;
385  }
386 
387  // Now check if the transition occurs after the given match. The
388  // untilDateTime of the current match uses the same UTC offsets as the
389  // transitionTime of the current transition, so no complicated adjustments
390  // are needed. We just make sure we compare 'w' with 'w', 's' with 's',
391  // and 'u' with 'u'.
392  const AtcDateTuple *match_until = &match->until_dt;
393  const AtcDateTuple *transition_time;
394  if (match_until->suffix == kAtcSuffixS) {
395  transition_time = tts;
396  } else if (match_until->suffix == kAtcSuffixU) {
397  transition_time = ttu;
398  } else { // assume 'w'
399  transition_time = ttw;
400  }
401  if (atc_date_tuple_compare(transition_time, match_until) < 0) {
402  return kAtcMatchStatusWithinMatch;
403  }
404 
405  return kAtcMatchStatusFarFuture;
406 }
