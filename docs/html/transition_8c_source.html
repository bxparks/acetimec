15 int8_t atc_date_tuple_compare(
16  const AtcDateTuple *a,
17  const AtcDateTuple *b)
18 {
19  if (a->year < b->year) return -1;
20  if (a->year > b->year) return 1;
21  if (a->month < b->month) return -1;
22  if (a->month > b->month) return 1;
23  if (a->day < b->day) return -1;
24  if (a->day > b->day) return 1;
25  if (a->minutes < b->minutes) return -1;
26  if (a->minutes > b->minutes) return 1;
27  return 0;
28 }

30 atc_time_t atc_date_tuple_subtract(
31  const AtcDateTuple *a,
32  const AtcDateTuple *b)
33 {
34  int32_t da = atc_local_date_to_epoch_days(a->year, a->month, a->day);
35  int32_t db = atc_local_date_to_epoch_days(b->year, b->month, b->day);
36 
37  // Subtract the days, before converting to seconds, to avoid overflowing the
38  // int32_t when a.year or b.year is more than 68 years from the
39  // atc_current_epoch_year.
40  return (da - db) * 86400 + (a->minutes - b->minutes) * 60;
41 }

44  const AtcDateTuple *tt,
45  int16_t offset_minutes,
46  int16_t delta_minutes,
47  AtcDateTuple *ttw,
48  AtcDateTuple *tts,
49  AtcDateTuple *ttu)

51 
52  if (tt->suffix == kAtcSuffixS) {

71  tts->day = tt->day;

73  tts->suffix = kAtcSuffixS;

75  ttw->year = tt->year;
76  ttw->month = tt->month;
77  ttw->day = tt->day;
78  ttw->minutes = (int16_t) (tt->minutes + (offset_minutes + delta_minutes));
79  ttw->suffix = kAtcSuffixW;
80  } else {
81  // Explicit set the suffix to 'w' in case it was something else.
82  *ttw = *tt;
83  ttw->suffix = kAtcSuffixW;
84 
85  tts->year = tt->year;
86  tts->month = tt->month;
87  tts->day = tt->day;

89  tts->suffix = kAtcSuffixS;
90

92  ttu->month = tt->month;
93  ttu->day = tt->day;
94  ttu->minutes = (int16_t) (tt->minutes - (delta_minutes + offset_minutes));
95  ttu->suffix = kAtcSuffixU;

103 void atc_date_tuple_normalize(AtcDateTuple *dt)
104 {
105  const int16_t kOneDayAsMinutes = 60 * 24;
106 
107  if (dt->minutes <= -kOneDayAsMinutes) {
108  atc_local_date_decrement_one_day(&dt->year, &dt->month, &dt->day);
109  dt->minutes += kOneDayAsMinutes;
110  } else if (kOneDayAsMinutes <= dt->minutes) {
111  atc_local_date_increment_one_day(&dt->year, &dt->month, &dt->day);
112  dt->minutes -= kOneDayAsMinutes;
113  } else {
114  // do nothing
115  }

118 uint8_t atc_date_tuple_compare_fuzzy(
119  const AtcDateTuple *t,
120  const AtcDateTuple *start,
121  const AtcDateTuple *until)
122 {
123  // Use int32_t because a delta year of 2730 or greater will exceed
124  // the range of an int16_t.
125  int32_t t_months = t->year * (int32_t) 12 + t->month;
126  int32_t start_months = start->year * (int32_t) 12 + start->month;
127  if (t_months < start_months - 1) return kAtcMatchStatusPrior;
128  int32_t until_months = until->year * 12 + until->month;
129  if (until_months + 1 < t_months) return kAtcMatchStatusFarFuture;
130  return kAtcMatchStatusWithinMatch;
131 }

261 AtcTransition *atc_transition_storage_add_active_candidates_to_active_pool(
262  AtcTransitionStorage *ts)
263 {
264  // Shift active candidates to the left into the Active pool.
265  uint8_t i_active = ts->index_prior;
266  uint8_t i_candidate = ts->index_candidate;
267  for (; i_candidate < ts->index_free; i_candidate++) {
268  if (is_match_status_active(ts->transitions[i_candidate]->match_status)) {
269  if (i_active != i_candidate) {
270  // Perform swap of pointers to AtcTransition.
271  AtcTransition *tmp = ts->transitions[i_active];
272  ts->transitions[i_active] = ts->transitions[i_candidate];
273  ts->transitions[i_candidate] = tmp;
274  }
275  ++i_active;
276  }
277  }
278 
279  ts->index_prior = i_active;
280  ts->index_candidate = i_active;
281  ts->index_free = i_active;
282 
283  return ts->transitions[i_active - 1];
284 }

306 const char *atc_transition_extract_letter(const AtcTransition *t)
307 {
308  // RULES column is '-' or hh:mm, so return NULL to indicate this.
309  if (t->rule == NULL) {
310  return NULL;
311  }
312 
313  // RULES point to a named rule, and LETTER is a single, printable character.
314  // Return the letter_buf which contains a NUL-terminated string containing the
315  // single character, as initialized in
316  // atc_processor_create_transition_for_year().
317  char letter = t->rule->letter;
318  if (letter >= 32) {
319  return t->letter_buf;
320  }
321 
322  // RULES points to a named rule, and the LETTER is a string. The
323  // rule->letter is a non-printable number < 32, which is an index into
324  // a list of strings given by match->era->zonePolicy->letters[].
325  const AtcZonePolicy *policy = t->match->era->zone_policy;
326  uint8_t num_letters = policy->num_letters;
327  if (letter >= num_letters) {
328  // This should never happen unless there is a programming error. If it
329  // does, return an empty string. (createTransitionForYear() sets
330  // letterBuf to a NUL terminated empty string if rule->letter < 32)
331  return t->letter_buf;
332  }
333 
334  // Return the string at index 'rule->letter'.
335  return policy->letters[(uint8_t) letter];
336 }

340 uint8_t atc_transition_compare_to_match(
341  const AtcTransition *t, const AtcMatchingEra *match)
342 {
343  // Find the previous Match offsets.
344  int16_t prev_match_offset_minutes;
345  int16_t prev_match_delta_minutes;
346  if (match->prev_match) {
347  prev_match_offset_minutes = match->prev_match->last_offset_minutes;
348  prev_match_delta_minutes = match->prev_match->last_delta_minutes;
349  } else {
350  prev_match_offset_minutes = atc_zone_era_std_offset_minutes(match->era);
351  prev_match_delta_minutes = 0;
352  }
353 
354  // Expand start times.
355  AtcDateTuple stw;
356  AtcDateTuple sts;
357  AtcDateTuple stu;
358  atc_date_tuple_expand(
359  &match->start_dt,
360  prev_match_offset_minutes,
361  prev_match_delta_minutes,
362  &stw,
363  &sts,
364  &stu);
365 
366  // Transition times.
367  const AtcDateTuple *ttw = &t->transition_time;
368  const AtcDateTuple *tts = &t->transition_time_s;
369  const AtcDateTuple *ttu = &t->transition_time_u;
370 
371  // Compare Transition to Match, where equality is assumed if *any* of the
372  // 'w', 's', or 'u' versions of the DateTuple are equal. This prevents
373  // duplicate Transition instances from being created in a few cases.
374  if (atc_date_tuple_compare(ttw, &stw) == 0
375  || atc_date_tuple_compare(tts, &sts) == 0
376  || atc_date_tuple_compare(ttu, &stu) == 0) {
377  return kAtcMatchStatusExactMatch;
378  }
379 
380  if (atc_date_tuple_compare(ttu, &stu) < 0) {
381  return kAtcMatchStatusPrior;
382  }
383 
384  // Now check if the transition occurs after the given match. The
385  // untilDateTime of the current match uses the same UTC offsets as the
386  // transitionTime of the current transition, so no complicated adjustments
387  // are needed. We just make sure we compare 'w' with 'w', 's' with 's',
388  // and 'u' with 'u'.
389  const AtcDateTuple *match_until = &match->until_dt;
390  const AtcDateTuple *transition_time;
391  if (match_until->suffix == kAtcSuffixS) {
392  transition_time = tts;
393  } else if (match_until->suffix == kAtcSuffixU) {
394  transition_time = ttu;
395  } else { // assume 'w'
396  transition_time = ttw;
397  }
398  if (atc_date_tuple_compare(transition_time, match_until) < 0) {
399  return kAtcMatchStatusWithinMatch;
400  }
401 
402  return kAtcMatchStatusFarFuture;
403 }

416 static void atc_calculate_fold_and_overlap(
417  uint8_t* fold,
418  uint8_t* num,
419  const AtcTransition* prev,
420  const AtcTransition* curr,
421  const AtcTransition* next,
422  atc_time_t epoch_seconds)
423 {
424  if (curr == NULL) {
425  *fold = 0;
426  *num = 0;
427  return;
428  }
429 
430  // Check if within forward overlap shadow from prev
431  bool is_overlap;
432  if (prev == NULL) {
433  is_overlap = false;
434  } else {
435  // Extract the shift from prev transition. Can be 0 in some cases where
436  // the zone changed from DST of one zone to the STD into another zone,
437  // causing the overall UTC offset to remain unchanged.
438  atc_time_t shift_seconds = atc_date_tuple_subtract(
439  &curr->start_dt, &prev->until_dt);
440  if (shift_seconds >= 0) {
441  // spring forward, or unchanged
442  is_overlap = false;
443  } else {
444  is_overlap = epoch_seconds - curr->start_epoch_seconds < -shift_seconds;
445  }
446  }
447  if (is_overlap) {
448  *fold = 1; // selects the second match
449  *num = 2;
450  return;
451  }
452 
453  // Check if within backward overlap shawdow from next
454  if (next == NULL) {
455  is_overlap = false;
456  } else {
457  // Extract the shift to next transition. Can be 0 in some cases where
458  // the zone changed from DST of one zone to the STD into another zone,
459  // causing the overall UTC offset to remain unchanged.
460  atc_time_t shift_seconds = atc_date_tuple_subtract(
461  &next->start_dt, &curr->until_dt);
462  if (shift_seconds >= 0) {
463  // spring forward, or unchanged
464  is_overlap = false;
465  } else {
466  // Check if within the backward overlap shadow from next
467  is_overlap = next->start_epoch_seconds - epoch_seconds <= -shift_seconds;
468  }
469  }
470  if (is_overlap) {
471  *fold = 0; // epochSeconds selects the first match
472  *num = 2;
473  return;
474  }
475 
476  // Normal single match, no overlap.
477  *fold = 0;
478  *num = 1;
479 }

481 AtcTransitionForSeconds atc_transition_storage_find_for_seconds(
482  const AtcTransitionStorage *ts,
483  atc_time_t epoch_seconds)
484 {
485  const AtcTransition *prev = NULL;
486  const AtcTransition *curr = NULL;
487  const AtcTransition *next = NULL;
488  for (uint8_t i = 0; i < ts->index_free; i++) {
489  next = ts->transitions[i];
490  if (next->start_epoch_seconds > epoch_seconds) break;
491  prev = curr;
492  curr = next;
493  next = NULL;
494  }
495 
496  uint8_t fold;
497  uint8_t num;
498  atc_calculate_fold_and_overlap(&fold, &num, prev, curr, next, epoch_seconds);
499  AtcTransitionForSeconds result = {curr, fold, num};
500  return result;

503 AtcTransitionForDateTime atc_transition_storage_find_for_date_time(
504  const AtcTransitionStorage *ts,
505  const AtcLocalDateTime *ldt)
506 {
507  // Convert LocalDateTime to DateTuple.
508  AtcDateTuple local_dt = {
509  ldt->year,
510  ldt->month,
511  ldt->day,
512  (int16_t) (ldt->hour * 60 + ldt->minute),
513  kAtcSuffixW
514  };
515 
516  // Examine adjacent pairs of Transitions, looking for an exact match, gap,
517  // or overlap.
518  const AtcTransition *prev = NULL;
519  const AtcTransition *curr = NULL;
520  uint8_t num = 0;
521  for (uint8_t i = 0; i < ts->index_free; i++) {
522  curr = ts->transitions[i];
523 
524  const AtcDateTuple *start_dt = &curr->start_dt;
525  const AtcDateTuple *until_dt = &curr->until_dt;
526  bool is_exact_match = atc_date_tuple_compare(start_dt, &local_dt) <= 0
527  && atc_date_tuple_compare(&local_dt, until_dt) < 0;
528 
529  if (is_exact_match) {
530  // Check for a previous exact match to detect an overlap.
531  if (num == 1) {
532  num++;
533  break;
534  }
535 
536  // Loop again to detect an overlap.
537  num = 1;
538  } else if (atc_date_tuple_compare(start_dt, &local_dt) > 0) {
539  // Exit loop since no more curr transition.
540  break;
541  }
542 
543  prev = curr;
544 
545  // Set nullptr so that if the loop runs off the end of the list of
546  // Transitions, the current transition is marked as nullptr.
547  curr = NULL;
548  }
549 
550  // If the prev was an exact match, set curr to the same to avoid confusion.
551  if (num == 1) {
552  curr = prev;
553  }
554 
555  AtcTransitionForDateTime result = {prev, curr, num};
556  return result;
557 }
