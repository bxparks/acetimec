18 int8_t atc_compare_era_to_year_month(
19  const AtcZoneEra *era,
20  int16_t year,
21  uint8_t month)
22 {
23  if (era->until_year < year) return -1;
24  if (era->until_year > year) return 1;
25  if (era->until_month < month) return -1;
26  if (era->until_month > month) return 1;
27  if (era->until_day > 1) return 1;
28  //if (era->until_time_minutes < 0) return -1; // never possible
29  if (era->until_time_code > 0) return 1;
30  return 0;
31 }

62 void atc_create_matching_era(
63  AtcMatchingEra *new_match,
64  AtcMatchingEra *prev_match,
65  const AtcZoneEra *era,
66  AtcYearMonth start_ym,
67  AtcYearMonth until_ym) {
68 
69  // If prev_match is null, set start_date to be earlier than all valid
70  // ZoneEra.
71  AtcDateTuple start_date;
72  if (prev_match == NULL) {
73  start_date.year = kAtcInvalidYear;
74  start_date.month = 1;
75  start_date.day = 1;
76  start_date.minutes = 0;
77  start_date.suffix = kAtcSuffixW;
78  } else {
79  start_date.year = prev_match->era->until_year;
80  start_date.month = prev_match->era->until_month;
81  start_date.day = prev_match->era->until_day;
82  start_date.minutes = atc_zone_era_until_minutes(prev_match->era);
83  start_date.suffix = atc_zone_era_until_suffix(prev_match->era);
84  }
85  AtcDateTuple lower_bound = {
86  start_ym.year,
87  start_ym.month,
88  1,
89  0,
90  kAtcSuffixW
91  };
92  if (atc_date_tuple_compare(&start_date, &lower_bound) < 0) {
93  start_date = lower_bound;
94  }
95 
96  AtcDateTuple until_date = {
97  era->until_year,
98  era->until_month,
99  era->until_day,
100  atc_zone_era_until_minutes(era),
101  atc_zone_era_until_suffix(era),
102  };
103  AtcDateTuple upper_bound = {
104  until_ym.year,
105  until_ym.month,
106  1,
107  0,
108  kAtcSuffixW
109  };
110  if (atc_date_tuple_compare(&upper_bound, &until_date) < 0) {
111  until_date = upper_bound;
112  }
113 
114  new_match->start_dt = start_date;
115  new_match->until_dt = until_date;
116  new_match->era = era;
117  new_match->prev_match = prev_match;
118  new_match->last_offset_minutes = 0;
119  new_match->last_delta_minutes = 0;
120 }

122 AtcMonthDay atc_processing_calc_start_day_of_month(
123  int16_t year,
124  uint8_t month,
125  uint8_t on_day_of_week,
126  int8_t on_day_of_month)
127 {
128  AtcMonthDay md;
129 
130  if (on_day_of_week == 0) {
131  md.month = month;
132  md.day = on_day_of_month;
133  return md;
134  }
135 
136  if (on_day_of_month >= 0) {
137  uint8_t days_in_month = atc_local_date_days_in_year_month(year, month);
138  if (on_day_of_month == 0) {
139  on_day_of_month = days_in_month - 6;
140  }
141  uint8_t dow = atc_local_date_day_of_week(year, month, on_day_of_month);
142  uint8_t day_of_week_shift = (on_day_of_week - dow + 7) % 7;
143  uint8_t day = (uint8_t) (on_day_of_month + day_of_week_shift);
144  if (day > days_in_month) {
145  // TODO: Support shifting from Dec to Jan of following year.
146  day -= days_in_month;
147  month++;
148  }
149  md.month = month;
150  md.day = day;
151  return md;
152  } else {
153  on_day_of_month = -on_day_of_month;
154  uint8_t dow = atc_local_date_day_of_week(year, month, on_day_of_month);
155  int8_t day_of_week_shift = (dow - on_day_of_week + 7) % 7;
156  int8_t day = on_day_of_month - day_of_week_shift;
157  if (day < 1) {
158  // TODO: Support shifting from Jan to Dec of the previous year.
159  month--;
160  uint8_t days_in_prev_month = atc_local_date_days_in_year_month(
161  year, month);
162  day += days_in_prev_month;
163  }
164  md.month = month;
165  md.day = day;
166  return md;
167  }
168 }

174 uint8_t atc_processing_find_matches(
175  const AtcZoneInfo *zone_info,
176  AtcYearMonth start_ym,
177  AtcYearMonth until_ym,
178  AtcMatchingEra *matches,
179  uint8_t num_matches)
180 {
181  uint8_t i_match = 0;
182  AtcMatchingEra *prev_match = NULL;
183  // Support both normal Zones and Links using accessors.
184  uint8_t num_eras = atc_zone_info_num_eras(zone_info);
185  for (uint8_t i_era = 0; i_era < num_eras; i_era++) {
186  const AtcZoneEra *era = atc_zone_info_era(zone_info, i_era);
187  if (atc_era_overlaps_interval(prev_match, era, start_ym, until_ym)) {
188  if (i_match < num_matches) {
189  atc_create_matching_era(
190  &matches[i_match], prev_match, era, start_ym, until_ym);
191  prev_match = &matches[i_match];
192  i_match++;
193  }
194  }
195  }
196  return i_match;
197 }

221 void atc_processing_create_transition_for_year(
222  AtcTransition *t,
223  int16_t year,
224  const AtcZoneRule *rule,
225  const AtcMatchingEra *match)
226 {
227  t->match = match;
228  t->rule = rule;
229  t->offset_minutes = atc_zone_era_std_offset_minutes(match->era);
230  t->letter_buf[0] = '\0';
231 
232  if (rule) {
233  atc_processing_get_transition_time(year, rule, &t->transition_time);
234  t->delta_minutes = atc_zone_rule_dst_offset_minutes(rule);
235  char letter = rule->letter;
236  if (letter >= 32) {
237  // If LETTER is a '-', treat it the same as an empty string.
238  if (letter != '-') {
239  t->letter_buf[0] = letter;
240  t->letter_buf[1] = '\0';
241  }
242  } else {
243  // rule->letter is a long string, so is referenced as an offset index into
244  // the ZonePolicy.letters array. The string cannot fit in letter_buf, so
245  // will be retrieved by the atc_processing_long_letter() function.
246  }
247  } else {
248  // Create a Transition using the MatchingEra for the transitionTime.
249  // Used for simple MatchingEra.
250  t->transition_time = match->start_dt;
251  t->delta_minutes = atc_zone_era_dst_offset_minutes(match->era);
252  }

307 void atc_processing_find_candidate_transitions(
308  AtcTransitionStorage *ts,
309  AtcMatchingEra *match)
310 {
311  const AtcZonePolicy *policy = match->era->zone_policy;
312  uint8_t num_rules = policy->num_rules;
313  int16_t start_year = match->start_dt.year;
314  int16_t end_year = match->until_dt.year;
315 
316  AtcTransition **prior = atc_transition_storage_reserve_prior(ts);
317  (*prior)->is_valid_prior = false;
318  for (uint8_t r = 0; r < num_rules; r++) {
319  const AtcZoneRule *rule = &policy->rules[r];
320 
321  // Add transitions for interior years
322  int16_t interior_years[kAtcMaxInteriorYears];
323  uint8_t num_years = atc_processing_calc_interior_years(
324  interior_years,
325  kAtcMaxInteriorYears,
326  rule->from_year,
327  rule->to_year,
328  start_year,
329  end_year);
330  for (uint8_t y = 0; y < num_years; y++) {
331  int16_t year = interior_years[y];
332  AtcTransition *t = atc_transition_storage_get_free_agent(ts);
333  atc_processing_create_transition_for_year(t, year, rule, match);
334  uint8_t status = atc_transition_compare_to_match_fuzzy(t, match);
335  if (status == kAtcMatchStatusPrior) {
336  atc_transition_storage_set_free_agent_as_prior_if_valid(ts);
337  } else if (status == kAtcMatchStatusWithinMatch) {
338  atc_transition_storage_add_free_agent_to_candidate_pool(ts);
339  } else {
340  // Must be kFarFuture.
341  // Do nothing, allowing the free agent to be reused.
342  }
343  }
344 
345  // Add Transition for prior year
346  int16_t prior_year = atc_processing_get_most_recent_prior_year(
347  rule->from_year, rule->to_year,
348  start_year, end_year);
349  if (prior_year != kAtcInvalidYear) {
350  AtcTransition *t = atc_transition_storage_get_free_agent(ts);
351  atc_processing_create_transition_for_year(t, prior_year, rule, match);
352  atc_transition_storage_set_free_agent_as_prior_if_valid(ts);
353  }
354  }
355 
356  // Add the reserved prior into the Candidate pool only if 'isValidPrior' is
357  // true.
358  if ((*prior)->is_valid_prior) {
359  atc_transition_storage_add_prior_to_candidate_pool(ts);
360  }
361 }

367 void atc_processing_process_transition_match_status(
368  AtcTransition *transition,
369  AtcTransition **prior)
370 {
371  uint8_t status = atc_transition_compare_to_match(
372  transition, transition->match);
373  transition->match_status = status;
374 
375  if (status == kAtcMatchStatusExactMatch) {
376  if (*prior) {
377  (*prior)->match_status = kAtcMatchStatusFarPast;
378  }
379  (*prior) = transition;
380  } else if (status == kAtcMatchStatusPrior) {
381  if (*prior) {
382  if (atc_date_tuple_compare(
383  &(*prior)->transition_time_u,
384  &transition->transition_time_u) <= 0) {
385  (*prior)->match_status = kAtcMatchStatusFarPast;
386  (*prior) = transition;
387  } else {
388  transition->match_status = kAtcMatchStatusFarPast;
389  }
390  } else {
391  (*prior) = transition;
392  }
393  }
394 }

473 //---------------------------------------------------------------------------
474 void atc_processing_generate_start_until_times(
475  AtcTransition **begin,
476  AtcTransition **end)
477 {
478  AtcTransition *prev = *begin;
479  bool is_after_first = false;
480 
481  for (AtcTransition **iter = begin; iter != end; ++iter) {
482  AtcTransition * const t = *iter;
483 
484  // 1) Update the untilDateTime of the previous Transition
485  const AtcDateTuple *tt = &t->transition_time;
486  if (is_after_first) {
487  prev->until_dt = *tt;
488  }
489 
490  // 2) Calculate the current startDateTime by shifting the
491  // transitionTime (represented in the UTC offset of the previous
492  // transition) into the UTC offset of the *current* transition.
493  int16_t minutes = tt->minutes + (
494  - prev->offset_minutes - prev->delta_minutes
495  + t->offset_minutes + t->delta_minutes);
496  t->start_dt.year = tt->year;
497  t->start_dt.month = tt->month;
498  t->start_dt.day = tt->day;
499  t->start_dt.minutes = minutes;
500  t->start_dt.suffix = tt->suffix;
501  atc_date_tuple_normalize(&t->start_dt);
502 
503  // 3) The epochSecond of the 'transitionTime' is determined by the
504  // UTC offset of the *previous* Transition. However, the
505  // transitionTime can be represented by an illegal time (e.g. 24:00).
506  // So, it is better to use the properly normalized startDateTime
507  // (calculated above) with the *current* UTC offset.
508  //
509  // NOTE: We should also be able to calculate this directly from
510  // 'transitionTimeU' which should still be a valid field, because it
511  // hasn't been clobbered by 'untilDateTime' yet. Not sure if this saves
512  // any CPU time though, since we still need to mutiply by 900.
513  const AtcDateTuple *st = &t->start_dt;
514  const atc_time_t offset_seconds = (atc_time_t) 60
515  * (st->minutes - (t->offset_minutes + t->delta_minutes));
516  int32_t epoch_seconds = (int32_t) 86400 * atc_local_date_to_epoch_days(
517  st->year, st->month, st->day);
518  t->start_epoch_seconds = epoch_seconds + offset_seconds;
519 
520  prev = t;
521  is_after_first = true;
522  }
523 
524  // The last Transition's until time is the until time of the MatchingEra.
525  AtcDateTuple until_time_w;
526  AtcDateTuple until_time_s;
527  AtcDateTuple until_time_u;
528  atc_date_tuple_expand(
529  &prev->match->until_dt,
530  prev->offset_minutes,
531  prev->delta_minutes,
532  &until_time_w,
533  &until_time_s,
534  &until_time_u);
535  prev->until_dt = until_time_w;
536 }

543 void atc_processing_create_abbreviation(
544  char* dest,
545  uint8_t dest_size,
546  const char* format,
547  uint16_t delta_minutes,
548  const char* letter_string) {
549 
550  // Check if FORMAT contains a '%'.
551  if (strchr(format, '%') != NULL) {
552  // Check if RULES column empty, therefore no 'letter'
553  if (letter_string == NULL) {
554  strncpy(dest, format, dest_size - 1);
555  dest[dest_size - 1] = '\0';
556  } else {
557  atc_copy_replace_string(
558  dest, dest_size, format, '%', letter_string);
559  }
560  } else {
561  // Check if FORMAT contains a '/'.
562  const char* slash_pos = strchr(format, '/');
563  if (slash_pos != NULL) {
564  if (delta_minutes == 0) {
565  uint8_t head_length = (slash_pos - format);
566  if (head_length >= dest_size) head_length = dest_size - 1;
567  memcpy(dest, format, head_length);
568  dest[head_length] = '\0';
569  } else {
570  uint8_t tail_length = strlen(slash_pos+1);
571  if (tail_length >= dest_size) tail_length = dest_size - 1;
572  memcpy(dest, slash_pos+1, tail_length);
573  dest[tail_length] = '\0';
574  }
575  } else {
576  // Just copy the FORMAT disregarding delta_minutes and letter_string.
577  strncpy(dest, format, dest_size);
578  dest[dest_size - 1] = '\0';
579  }
580  }
581 }

616 int8_t atc_processing_init_for_year(
617  AtcZoneProcessing *processing,
618  const AtcZoneInfo *zone_info,
619  int16_t year)
620 {
621  if (processing->zone_info != zone_info) {
622  atc_processing_init(processing);
623  processing->zone_info = zone_info;
624  }
625  if (atc_processing_is_filled_for_year(processing, year)) return kAtcErrOk;
626 
627  processing->year = year;
628  processing->num_matches = 0;
629  atc_transition_storage_init(&processing->transition_storage);
630  const AtcZoneContext *context = processing->zone_info->zone_context;
631  if (year < context->start_year - 1 || context->until_year < year) {
632  return kAtcErrGeneric;
633  }
634  AtcYearMonth start_ym = { year - 1, 12 };
635  AtcYearMonth until_ym = { year + 1, 2 };
636 
637  // Step 1: Find matches.
638  uint8_t num_matches = atc_processing_find_matches(
639  processing->zone_info,
640  start_ym,
641  until_ym,
642  processing->matches,
643  kAtcMaxMatches);
644 
645  // Step 2: Create Transitions.
646  atc_processing_create_transitions(
647  &processing->transition_storage,
648  processing->matches,
649  num_matches);
650 
651  // Step 3: Fix transition times.
652  AtcTransitionStorage *ts = &processing->transition_storage;
653  AtcTransition **begin = &ts->transitions[0];
654  AtcTransition **end = &ts->transitions[ts->index_free];
655  atc_transition_fix_times(begin, end);
656 
657  // Step 4: Generate start and until times.
658  atc_processing_generate_start_until_times(begin, end);
659 
660  // Step 5: Calc abbreviations.
661  atc_processing_calc_abbreviations(begin, end);
662 
663  return kAtcErrOk;

693 static uint8_t atc_processing_calculate_fold(
694  atc_time_t epoch_seconds,
695  const AtcTransition *match,
696  const AtcTransition *prev_match)
697 {
698  if (match == NULL) return 0;
699  if (prev_match == NULL) return 0;
700 
701  // Check if epoch_seconds occurs during a "fall back" DST transition.
702  atc_time_t overlap_seconds = atc_date_tuple_subtract(
703  &prev_match->until_dt, &match->start_dt);
704  if (overlap_seconds <= 0) return 0;
705  atc_time_t seconds_from_transition_start =
706  epoch_seconds - match->start_epoch_seconds;
707  if (seconds_from_transition_start >= overlap_seconds) return 0;
708 
709  // EpochSeconds occurs within the "fall back" overlap.
710  return 1;
711 }

713 static AtcMatchingTransition atc_processing_find_transition_for_seconds(
714  const AtcTransitionStorage *ts,
715  atc_time_t epoch_seconds)
716 {
717  const AtcTransition *prev_match = NULL;
718  const AtcTransition *match = NULL;
719  for (uint8_t i = 0; i < ts->index_free; i++) {
720  const AtcTransition *candidate = ts->transitions[i];
721  if (candidate->start_epoch_seconds > epoch_seconds) break;
722  prev_match = match;
723  match = candidate;
724  }
725  uint8_t fold = atc_processing_calculate_fold(
726  epoch_seconds, match, prev_match);
727  AtcMatchingTransition result = { match, fold };
728  return result;
729 }

731 static AtcTransitionResult atc_processing_find_transition_for_date_time(
732  const AtcTransitionStorage *ts,
733  const AtcLocalDateTime *ldt)
734 {
735  // Convert LocalDateTime to DateTuple.
736  AtcDateTuple local_dt = {
737  ldt->year,
738  ldt->month,
739  ldt->day,
740  (int16_t) (ldt->hour * 60 + ldt->minute),
741  kAtcSuffixW
742  };
743 
744  // Examine adjacent pairs of Transitions, looking for an exact match, gap,
745  // or overlap.
746  const AtcTransition *prev_candidate = NULL;
747  const AtcTransition *candidate = NULL;
748  int8_t search_status = kAtcSearchStatusGap;
749  for (uint8_t i = 0; i < ts->index_free; i++) {
750  candidate = ts->transitions[i];
751 
752  const AtcDateTuple *start_dt = &candidate->start_dt;
753  const AtcDateTuple *until_dt = &candidate->until_dt;
754  bool is_exact_match = atc_date_tuple_compare(start_dt, &local_dt) <= 0
755  && atc_date_tuple_compare(&local_dt, until_dt) < 0;
756 
757  if (is_exact_match) {
758  // Check for a previous exact match to detect an overlap.
759  if (search_status == kAtcSearchStatusExact) {
760  search_status = kAtcSearchStatusOverlap;
761  break;
762  }
763 
764  // Loop again to detect an overlap.
765  search_status = kAtcSearchStatusExact;
766 
767  } else if (atc_date_tuple_compare(start_dt, &local_dt) > 0) {
768  // Exit loop since no more candidate transition.
769  break;
770  }
771 
772  prev_candidate = candidate;
773 
774  // Set nullptr so that if the loop runs off the end of the list of
775  // Transitions, the candidate is marked as nullptr.
776  candidate = NULL;
777  }
778 
779  // Check if the prev was an exact match, and clear the current to
780  // avoid confusion.
781  if (search_status == kAtcSearchStatusExact) {
782  candidate = NULL;
783  }
784 
785  AtcTransitionResult result = {
786  prev_candidate,
787  candidate,
788  search_status,
789  };
790  return result;
791 }

846 // Adapted from ExtendedZoneProcessor::getOffsetDateTime(const LocalDatetime&)
847 // from ExtendedZoneProcessor in AceTime.
848 int8_t atc_processing_offset_date_time_from_local_date_time(
849  AtcZoneProcessing *processing,
850  const AtcZoneInfo *zone_info,
851  const AtcLocalDateTime *ldt,
852  uint8_t fold,
853  AtcOffsetDateTime *odt)
854 {
855  int8_t err = atc_processing_init_for_year(processing, zone_info, ldt->year);
856  if (err) return err;
857 
858  AtcTransitionResult result =
859  atc_processing_find_transition_for_date_time(
860  &processing->transition_storage, ldt);
861 
862  // Extract the appropriate Transition, depending on the requested 'fold'
863  // and the 'result.search_status'.
864  bool needs_normalization = false;
865  const AtcTransition *t;
866  if (result.search_status == kAtcSearchStatusExact) {
867  t = result.transition0;
868  } else {
869  if (result.transition0 == NULL || result.transition1 == NULL) {
870  // ldt was far past or far future, and didn't match anything.
871  t = NULL;
872  } else {
873  needs_normalization = (result.search_status == kAtcSearchStatusGap);
874  t = (fold == 0) ? result.transition0 : result.transition1;
875  }
876  }
877 
878  if (! t) return kAtcErrGeneric;
879 
880  odt->year = ldt->year;
881  odt->month = ldt->month;
882  odt->day = ldt->day;
883  odt->hour = ldt->hour;
884  odt->minute = ldt->minute;
885  odt->second = ldt->second;
886  odt->offset_minutes = t->offset_minutes + t->delta_minutes;
887  odt->fold = fold;
888 
889  if (needs_normalization) {
890  atc_time_t epoch_seconds = atc_offset_date_time_to_epoch_seconds(odt);
891 
892  // If in the gap, normalization means that we convert to epochSeconds
893  // then perform another search through the Transitions, then use
894  // that new Transition to convert the epochSeconds to OffsetDateTime. It
895  // turns out that this process identical to just using the other
896  // Transition returned in TransitionResult above.
897  const AtcTransition *othert = (fold == 0)
898  ? result.transition1
899  : result.transition0;
900  int8_t err = atc_offset_date_time_from_epoch_seconds(
901  epoch_seconds,
902  othert->offset_minutes + othert->delta_minutes,
903  odt);
904  if (err) return err;
905 
906  // Invert the fold.
907  // 1) The normalization process causes the LocalDateTime to jump to the
908  // other transition.
909  // 2) Provides a user-accessible indicator that a gap normalization was
910  // performed.
911  odt->fold = 1 - fold;
912  }
913 
914  return kAtcErrOk;
915 }
