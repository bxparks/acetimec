18 // Adapted from TimeZone::getOffsetDateTime(epochSeconds) from the
19 // AceTime library.
20 int8_t atc_time_zone_offset_date_time_from_epoch_seconds(
21  const AtcTimeZone *tz,
22  atc_time_t epoch_seconds,
23  AtcOffsetDateTime *odt)
24 {
25  int32_t offset_seconds;
26  uint8_t fold;
27  if (tz->zone_info) {
28  atc_processor_init_for_zone_info(tz->zone_processor, tz->zone_info);
29 
30  AtcFindResult result;
31  int8_t err = atc_processor_find_by_epoch_seconds(
32  tz->zone_processor, epoch_seconds, &result);
33  if (err) return err;
34  if (result.type == kAtcFindResultNotFound) return kAtcErrGeneric;
35  offset_seconds = result.std_offset_seconds + result.dst_offset_seconds;
36  fold = result.fold;
37  } else {
38  offset_seconds = 0;
39  fold = 0;
40  }
41 
42  int8_t err = atc_offset_date_time_from_epoch_seconds(
43  epoch_seconds, offset_seconds, odt);
44  if (err) return err;
45 
46  odt->fold = fold;
47  return kAtcErrOk;
48 }

50 // Adapted from TimeZone::getOffsetDateTime(const LocalDatetime&) from the

52 int8_t atc_time_zone_offset_date_time_from_local_date_time(
53  const AtcTimeZone *tz,
54  const AtcLocalDateTime *ldt,
55  AtcOffsetDateTime *odt)
56 {
57  if (tz->zone_info) {
58  atc_processor_init_for_zone_info(tz->zone_processor, tz->zone_info);
59 
60  AtcFindResult result;
61  int8_t err = atc_processor_find_by_local_date_time(
62  tz->zone_processor, ldt, &result);
63  if (err) return err;
64  if (result.type == kAtcFindResultNotFound) return kAtcErrGeneric;
65 
66  // Convert FindResult into OffsetDateTime using the requested offset.
67  odt->year = ldt->year;
68  odt->month = ldt->month;
69  odt->day = ldt->day;
70  odt->hour = ldt->hour;
71  odt->minute = ldt->minute;
72  odt->second = ldt->second;
73  odt->offset_seconds =
74  result.req_std_offset_seconds + result.req_dst_offset_seconds;
75  odt->fold = result.fold;
76 
77  // Special process for kAtcFindResultGap: Convert to epoch_seconds using the
78  // req_std_offset_seconds and req_dst_offset_seconds, then convert back to
79  // AtcOffsetDateTime using the target's std_offset_seconds and
80  // dst_offset_seconds.
81  if (result.type == kAtcFindResultGap) {
82  atc_time_t epoch_seconds = atc_offset_date_time_to_epoch_seconds(odt);
83  int32_t target_offset =
84  result.std_offset_seconds + result.dst_offset_seconds;
85  atc_offset_date_time_from_epoch_seconds(
86  epoch_seconds, target_offset, odt);
87  }
88  } else {
89  odt->year = ldt->year;
90  odt->month = ldt->month;
91  odt->day = ldt->day;
92  odt->hour = ldt->hour;
93  odt->minute = ldt->minute;
94  odt->second = ldt->second;
95  odt->offset_seconds = 0;
96  odt->fold = 0;
97  }
98 
99  return kAtcErrOk;
100 }

102 int8_t atc_time_zone_zoned_extra_from_epoch_seconds(
103  const AtcTimeZone *tz,
104  atc_time_t epoch_seconds,
105  AtcZonedExtra *extra)
106 {
107  if (epoch_seconds == kAtcInvalidEpochSeconds) return kAtcErrGeneric;
108 
109  if (tz->zone_info) {
110  atc_processor_init_for_zone_info(tz->zone_processor, tz->zone_info);
111 
112  AtcFindResult result;
113  int8_t err = atc_processor_find_by_epoch_seconds(
114  tz->zone_processor, epoch_seconds, &result);
115  if (err) return err;
116 
117  extra->type = result.type;
118  extra->std_offset_seconds = result.std_offset_seconds;
119  extra->dst_offset_seconds = result.dst_offset_seconds;
120  extra->req_std_offset_seconds = result.req_std_offset_seconds;
121  extra->req_dst_offset_seconds = result.req_dst_offset_seconds;
122  memcpy(extra->abbrev, result.abbrev, kAtcAbbrevSize);
123  extra->abbrev[kAtcAbbrevSize - 1] = '\0';
124  } else {
125  extra->type = kAtcFindResultExact;
126  extra->std_offset_seconds = 0;
127  extra->dst_offset_seconds = 0;
128  extra->req_std_offset_seconds = 0;
129  extra->req_dst_offset_seconds = 0;
130  memcpy(extra->abbrev, "UTC", sizeof("UTC"));
131  }
132 
133  return kAtcErrOk;
134 }
