45 void atc_print_uint16(AtcStringBuffer *sb, uint16_t n)
46 {
47  // Calculate the digits in little-endian order.
48  char tmp[5];
49  uint8_t i = 0;
50  do {
51  uint16_t digit = n % 10;
52  n /= 10;
53  tmp[i] = digit + '0';
54  i++;
55  } while (n != 0);
56 
57  // Print the digits in reversed, big-endian order for humans.
58  while (i--) {
59  atc_print_char(sb, tmp[i]);
60  }
61 }
