18 int8_t atc_compare_era_to_year_month(
19  const AtcZoneEra *era,
20  int16_t year,
21  uint8_t month)
22 {
23  if (era->until_year < year) return -1;
24  if (era->until_year > year) return 1;
25  if (era->until_month < month) return -1;
26  if (era->until_month > month) return 1;
27  if (era->until_day > 1) return 1;
28 
29  int32_t until_time_seconds = atc_zone_era_until_seconds(era);
30  // if (until_time_seconds < 0) return -1; // never possible
31  if (until_time_seconds > 0) return 1;
32  return 0;
33 }

78 void atc_create_matching_era(
79  AtcMatchingEra *new_match,
80  AtcMatchingEra *prev_match,
81  const AtcZoneEra *era,
82  AtcYearMonth start_ym,
83  AtcYearMonth until_ym) {
84 
85  // If prev_match is null, set start_date to be earlier than all valid
86  // ZoneEra.
87  AtcDateTuple start_date;
88  if (prev_match == NULL) {
89  start_date.year = kAtcInvalidYear;
90  start_date.month = 1;
91  start_date.day = 1;
92  start_date.seconds = 0;
93  start_date.suffix = kAtcSuffixW;
94  } else {
95  start_date.year = prev_match->era->until_year;
96  start_date.month = prev_match->era->until_month;
97  start_date.day = prev_match->era->until_day;
98  start_date.seconds = atc_zone_era_until_seconds(prev_match->era);
99  start_date.suffix = atc_zone_era_until_suffix(prev_match->era);
100  }
101  AtcDateTuple lower_bound = {
102  start_ym.year,
103  start_ym.month,
104  1,
105  0,
106  kAtcSuffixW
107  };
108  if (atc_date_tuple_compare(&start_date, &lower_bound) < 0) {
109  start_date = lower_bound;
110  }
111 
112  AtcDateTuple until_date = {
113  era->until_year,
114  era->until_month,
115  era->until_day,
116  atc_zone_era_until_seconds(era),
117  atc_zone_era_until_suffix(era),
118  };
119  AtcDateTuple upper_bound = {
120  until_ym.year,
121  until_ym.month,
122  1,
123  0,
124  kAtcSuffixW
125  };
126  if (atc_date_tuple_compare(&upper_bound, &until_date) < 0) {
127  until_date = upper_bound;
128  }
129 
130  new_match->start_dt = start_date;
131  new_match->until_dt = until_date;
132  new_match->era = era;
133  new_match->prev_match = prev_match;
134  new_match->last_offset_seconds = 0;
135  new_match->last_delta_seconds = 0;
136 }

138 AtcMonthDay atc_processor_calc_start_day_of_month(
139  int16_t year,
140  uint8_t month,
141  uint8_t on_day_of_week,
142  int8_t on_day_of_month)
143 {
144  AtcMonthDay md;
145 
146  if (on_day_of_week == 0) {
147  md.month = month;
148  md.day = on_day_of_month;
149  return md;
150  }
151 
152  if (on_day_of_month >= 0) {
153  uint8_t days_in_month = atc_local_date_days_in_year_month(year, month);
154  if (on_day_of_month == 0) {
155  on_day_of_month = days_in_month - 6;
156  }
157  uint8_t dow = atc_local_date_day_of_week(year, month, on_day_of_month);
158  uint8_t day_of_week_shift = (on_day_of_week - dow + 7) % 7;
159  uint8_t day = (uint8_t) (on_day_of_month + day_of_week_shift);
160  if (day > days_in_month) {
161  // TODO: Support shifting from Dec to Jan of following year.
162  day -= days_in_month;
163  month++;
164  }
165  md.month = month;
166  md.day = day;
167  } else {
168  on_day_of_month = -on_day_of_month;
169  uint8_t dow = atc_local_date_day_of_week(year, month, on_day_of_month);
170  int8_t day_of_week_shift = (dow - on_day_of_week + 7) % 7;
171  int8_t day = on_day_of_month - day_of_week_shift;
172  if (day < 1) {
173  // TODO: Support shifting from Jan to Dec of the previous year.
174  month--;
175  uint8_t days_in_prev_month = atc_local_date_days_in_year_month(
176  year, month);
177  day += days_in_prev_month;
178  }
179  md.month = month;
180  md.day = day;
181  }
182  return md;
183 }

189 uint8_t atc_processor_find_matches(
190  const AtcZoneInfo *zone_info,
191  AtcYearMonth start_ym,
192  AtcYearMonth until_ym,
193  AtcMatchingEra *matches,
194  uint8_t matches_size)
195 {
196  uint8_t i_match = 0;
197  AtcMatchingEra *prev_match = NULL;
198  uint8_t num_eras = zone_info->num_eras;
199  for (uint8_t i_era = 0; i_era < num_eras; i_era++) {
200  const AtcZoneEra *era = &zone_info->eras[i_era];
201  if (atc_era_overlaps_interval(
202  (prev_match ? prev_match->era : NULL),
203  era, start_ym, until_ym)) {
204  if (i_match < matches_size) {
205  atc_create_matching_era(
206  &matches[i_match], prev_match, era, start_ym, until_ym);
207  prev_match = &matches[i_match];
208  i_match++;
209  }
210  }
211  }
212  return i_match;
213 }

314 void atc_processor_find_candidate_transitions(
315  AtcTransitionStorage *ts,
316  AtcMatchingEra *match)
317 {
318  const AtcZonePolicy *policy = match->era->zone_policy;
319  uint8_t num_rules = policy->num_rules;
320  int16_t start_year = match->start_dt.year;
321  int16_t end_year = match->until_dt.year;
322 
323  AtcTransition **prior = atc_transition_storage_reserve_prior(ts);
324  (*prior)->is_valid_prior = false;
325  const char* const* letters = ts->zone_info->zone_context->letters;
326  for (uint8_t r = 0; r < num_rules; r++) {
327  const AtcZoneRule *rule = &policy->rules[r];
328 
329  // Add transitions for interior years
330  int16_t interior_years[kAtcMaxInteriorYears];
331  uint8_t num_years = atc_processor_calc_interior_years(
332  interior_years,
333  kAtcMaxInteriorYears,
334  rule->from_year,
335  rule->to_year,
336  start_year,
337  end_year);
338  for (uint8_t y = 0; y < num_years; y++) {
339  int16_t year = interior_years[y];
340  AtcTransition *t = atc_transition_storage_get_free_agent(ts);
341  atc_processor_create_transition_for_year(t, year, rule, match, letters);
342  uint8_t status = atc_transition_compare_to_match_fuzzy(t, match);
343  if (status == kAtcComparePrior) {
344  atc_transition_storage_set_free_agent_as_prior_if_valid(ts);
345  } else if (status == kAtcCompareWithinMatch) {
346  atc_transition_storage_add_free_agent_to_candidate_pool(ts);
347  } else {
348  // Must be kFarFuture.
349  // Do nothing, allowing the free agent to be reused.
350  }
351  }
352 
353  // Add Transition for prior year
354  int16_t prior_year = atc_processor_get_most_recent_prior_year(
355  rule->from_year, rule->to_year,
356  start_year, end_year);
357  if (prior_year != kAtcInvalidYear) {
358  AtcTransition *t = atc_transition_storage_get_free_agent(ts);
359  atc_processor_create_transition_for_year(
360  t, prior_year, rule, match, letters);
361  atc_transition_storage_set_free_agent_as_prior_if_valid(ts);
362  }
363  }
364 
365  // Add the reserved prior into the Candidate pool only if 'isValidPrior' is
366  // true.
367  if ((*prior)->is_valid_prior) {
368  atc_transition_storage_add_prior_to_candidate_pool(ts);
369  }
370 }

376 void atc_processor_process_transition_match_status(
377  AtcTransition *transition,
378  AtcTransition **prior)
379 {
380  uint8_t status = atc_transition_compare_to_match(
381  transition, transition->match);
382  transition->match_status = status;
383 
384  if (status == kAtcCompareExactMatch) {
385  if (*prior) {
386  (*prior)->match_status = kAtcCompareFarPast;
387  }
388  (*prior) = transition;
389  } else if (status == kAtcComparePrior) {
390  if (*prior) {
391  if (atc_date_tuple_compare(
392  &(*prior)->transition_time_u,
393  &transition->transition_time_u) <= 0) {
394  (*prior)->match_status = kAtcCompareFarPast;
395  (*prior) = transition;
396  } else {
397  transition->match_status = kAtcCompareFarPast;
398  }
399  } else {
400  (*prior) = transition;
401  }
402  }
403 }

480 //---------------------------------------------------------------------------
481 // Step 4
482 //---------------------------------------------------------------------------
483 void atc_processor_generate_start_until_times(
484  AtcTransition **begin,
485  AtcTransition **end)
486 {
487  AtcTransition *prev = *begin;
488  bool is_after_first = false;
489 
490  for (AtcTransition **iter = begin; iter != end; ++iter) {
491  AtcTransition * const t = *iter;
492 
493  // 1) Update the untilDateTime of the previous Transition
494  const AtcDateTuple *tt = &t->transition_time;
495  if (is_after_first) {
496  prev->until_dt = *tt;
497  }
498 
499  // 2) Calculate the current startDateTime by shifting the
500  // transitionTime (represented in the UTC offset of the previous
501  // transition) into the UTC offset of the *current* transition.
502  int32_t seconds = tt->seconds + (
503  - prev->offset_seconds - prev->delta_seconds
504  + t->offset_seconds + t->delta_seconds);
505  t->start_dt.year = tt->year;
506  t->start_dt.month = tt->month;
507  t->start_dt.day = tt->day;
508  t->start_dt.seconds = seconds;
509  t->start_dt.suffix = tt->suffix;
510  atc_date_tuple_normalize(&t->start_dt);
511 
512  // 3) The epochSecond of the 'transitionTime' is determined by the
513  // UTC offset of the *previous* Transition. However, the
514  // transitionTime can be represented by an illegal time (e.g. 24:00).
515  // So, it is better to use the properly normalized startDateTime
516  // (calculated above) with the *current* UTC offset.
517  //
518  // NOTE: We should also be able to calculate this directly from
519  // 'transitionTimeU' which should still be a valid field, because it
520  // hasn't been clobbered by 'untilDateTime' yet. Not sure if this saves
521  // any CPU time though, since we still need to mutiply by 900.
522  const AtcDateTuple *st = &t->start_dt;
523  const atc_time_t offset_seconds = (atc_time_t)
524  (st->seconds - (t->offset_seconds + t->delta_seconds));
525  atc_time_t epoch_seconds = (atc_time_t) 86400
526  * atc_local_date_to_epoch_days(st->year, st->month, st->day);
527  t->start_epoch_seconds = epoch_seconds + offset_seconds;
528 
529  prev = t;
530  is_after_first = true;
531  }
532 
533  // The last Transition's until time is the until time of the MatchingEra.
534  AtcDateTuple until_time_w;
535  AtcDateTuple until_time_s;
536  AtcDateTuple until_time_u;
537  atc_date_tuple_expand(
538  &prev->match->until_dt,
539  prev->offset_seconds,
540  prev->delta_seconds,
541  &until_time_w,
542  &until_time_s,
543  &until_time_u);
544  prev->until_dt = until_time_w;
545 }

552 void atc_processor_create_abbreviation(
553  char *dest,
554  uint8_t dest_size,
555  const char *format,
556  int32_t delta_seconds,
557  const char *letter_string) {
558 
559  // Check if FORMAT contains a '%'.
560  if (strchr(format, '%') != NULL) {
561  // Check if RULES column empty, therefore no 'letter'
562  if (letter_string == NULL) {
563  strncpy(dest, format, dest_size - 1);
564  dest[dest_size - 1] = '\0';
565  } else {
566  atc_copy_replace_string(
567  dest, dest_size, format, '%', letter_string);
568  }
569  } else {
570  // Check if FORMAT contains a '/'.
571  const char* slash_pos = strchr(format, '/');
572  if (slash_pos != NULL) {
573  if (delta_seconds == 0) {
574  uint8_t head_length = (slash_pos - format);
575  if (head_length >= dest_size) head_length = dest_size - 1;
576  memcpy(dest, format, head_length);
577  dest[head_length] = '\0';
578  } else {
579  uint8_t tail_length = strlen(slash_pos+1);
580  if (tail_length >= dest_size) tail_length = dest_size - 1;
581  memcpy(dest, slash_pos+1, tail_length);
582  dest[tail_length] = '\0';
583  }
584  } else {
585  // Just copy the FORMAT disregarding delta_seconds and letter_string.
586  strncpy(dest, format, dest_size);
587  dest[dest_size - 1] = '\0';
588  }
589  }
590 }

634 int8_t atc_processor_init_for_year(
635  AtcZoneProcessor *processor,
636  int16_t year)
637 {
638  if (atc_processor_is_filled_for_year(processor, year)) return kAtcErrOk;
639 
640  processor->year = year;
641  processor->num_matches = 0;
642  atc_transition_storage_init(
643  &processor->transition_storage, processor->zone_info);
644  const AtcZoneContext *context = processor->zone_info->zone_context;
645  if (year < context->start_year - 1 || context->until_year < year) {
646  return kAtcErrGeneric;
647  }
648  AtcYearMonth start_ym = { year - 1, 12 };
649  AtcYearMonth until_ym = { year + 1, 2 };
650 
651  // Step 1: Find matches.
652  uint8_t num_matches = atc_processor_find_matches(
653  processor->zone_info,
654  start_ym,
655  until_ym,
656  processor->matches,
657  kAtcMaxMatches);
658 
659  // Step 2: Create Transitions.
660  atc_processor_create_transitions(
661  &processor->transition_storage,
662  processor->matches,
663  num_matches);
664 
665  // Step 3: Fix transition times of active transitions.
666  AtcTransitionStorage *ts = &processor->transition_storage;
667  AtcTransition **begin = &ts->transitions[0];
668  AtcTransition **end = &ts->transitions[ts->index_prior];
669  atc_transition_fix_times(begin, end);
670 
671  // Step 4: Generate start and until times.
672  atc_processor_generate_start_until_times(begin, end);
673 
674  // Step 5: Calc abbreviations.
675  atc_processor_calc_abbreviations(begin, end);
676 
677  return kAtcErrOk;
678 }

695 void atc_processor_find_by_epoch_seconds(
696  AtcZoneProcessor *processor,
697  atc_time_t epoch_seconds,
698  AtcFindResult *result)
699 {
700  int8_t err = atc_processor_init_for_epoch_seconds(processor, epoch_seconds);
701  if (err) {
702  result->type = kAtcFindResultNotFound;
703  return;
704  }
705 
706  AtcTransitionForSeconds tfs = atc_transition_storage_find_for_seconds(
707  &processor->transition_storage, epoch_seconds);
708  const AtcTransition *t = tfs.curr;
709  if (! t) {
710  result->type = kAtcFindResultNotFound;
711  return;
712  }
713 
714  result->std_offset_seconds = t->offset_seconds;
715  result->dst_offset_seconds = t->delta_seconds;
716  result->req_std_offset_seconds = t->offset_seconds;
717  result->req_dst_offset_seconds = t->delta_seconds;
718  result->abbrev = t->abbrev;
719  result->fold = tfs.fold;
720  if (tfs.num == 2) {
721  result->type = kAtcFindResultOverlap;
722  } else {
723  result->type = kAtcFindResultExact;
724  }

727 // Adapted from ExtendedZoneProcessor::findByLocalDateTime() in the AceTime
728 // library.
729 void atc_processor_find_by_local_date_time(
730  AtcZoneProcessor *processor,
731  const AtcLocalDateTime *ldt,
732  AtcFindResult *result)
733 {
734 
735  int8_t err = atc_processor_init_for_year(processor, ldt->year);
736  if (err) {
737  result->type = kAtcFindResultNotFound;
738  return;
739  }
740 
741  AtcTransitionForDateTime tfd = atc_transition_storage_find_for_date_time(
742  &processor->transition_storage, ldt);
743 
744  // Extract the target Transition, depending on the requested fold
745  // and the tfd.num.
746  const AtcTransition *transition;
747  if (tfd.num == 1) {
748  transition = tfd.curr;
749  result->type = kAtcFindResultExact;
750  result->fold = 0;
751  result->req_std_offset_seconds = transition->offset_seconds;
752  result->req_dst_offset_seconds = transition->delta_seconds;
753  } else { // num = 0 or 2
754  if (tfd.prev == NULL || tfd.curr == NULL) {
755  // ldt was far past or far future
756  transition = NULL;
757  result->type = kAtcFindResultNotFound;
758  result->fold = 0;
759  } else { // gap or overlap
760  if (tfd.num == 0) { // num==0, Gap
761  result->type = kAtcFindResultGap;
762  result->fold = 0;
763  if (ldt->fold == 0) {
764  // ldt wants to use the 'prev' transition to convert to
765  // epochSeconds.
766  result->req_std_offset_seconds = tfd.prev->offset_seconds;
767  result->req_dst_offset_seconds = tfd.prev->delta_seconds;
768  // But after normalization, it will be shifted into the curr
769  // transition, so select 'curr' as the target transition.
770  transition = tfd.curr;
771  } else {
772  // ldt wants to use the 'curr' transition to convert to
773  // epochSeconds.
774  result->req_std_offset_seconds = tfd.curr->offset_seconds;
775  result->req_dst_offset_seconds = tfd.curr->delta_seconds;
776  // But after normalization, it will be shifted into the prev
777  // transition, so select 'prev' as the target transition.
778  transition = tfd.prev;
779  }
780  } else { // num==2, Overlap
781  transition = (ldt->fold == 0) ? tfd.prev : tfd.curr;
782  result->type = kAtcFindResultOverlap;
783  result->fold = ldt->fold;
784  result->req_std_offset_seconds = transition->offset_seconds;
785  result->req_dst_offset_seconds = transition->delta_seconds;
786  }
787  }
788  }
789 
790  if (! transition) {
791  result->type = kAtcFindResultNotFound;
792  return;
793  }
794 
795  result->std_offset_seconds = transition->offset_seconds;
796  result->dst_offset_seconds = transition->delta_seconds;
797  result->abbrev = transition->abbrev;
798 }
