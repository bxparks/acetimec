70 int32_t atc_convert_to_days(int16_t year, uint8_t month, uint8_t day)
71 {
72  int16_t year_prime = year - ((month <= 2) ? 1 : 0); // year begins on Mar 1
73  uint16_t era = year_prime / 400; // [0,24]
74  uint16_t year_of_era = year_prime - 400 * era; // [0,399]
75 
76  uint8_t month_prime = (month <= 2) ? month + 9 : month - 3; // [0,11]
77  uint16_t days_until_month_prime =
78  atc_convert_to_days_until_month_prime(month_prime);
79  uint16_t day_of_year_prime = days_until_month_prime + day - 1; // [0,365]
80  uint32_t day_of_era = (uint32_t) 365 * year_of_era + (year_of_era / 4)
81  - (year_of_era / 100) + day_of_year_prime; // [0,146096]
82 
83  // epoch_prime days is relative to 0000-03-01
84  int32_t day_of_epoch_prime = day_of_era + 146097 * era;
85  return day_of_epoch_prime
86  - (kAtcConverterEpochYear / 400) * 146097 // relative to 2000-03-01
87  + 60; // relative to 2000-01-01, 2000 is a leap year
88 }

90 void atc_convert_from_days(
91  int32_t epoch_days,
92  int16_t *year,
93  uint8_t *month,
94  uint8_t *day)
95 {
96  // epoch_prime days is relative to 0000-03-01
97  int32_t day_of_epoch_prime = epoch_days
98  + (kAtcConverterEpochYear / 400) * 146097
99  - 60;
100 
101  uint16_t era = (uint32_t) day_of_epoch_prime / 146097; // [0,24]
102  uint32_t day_of_era = day_of_epoch_prime - 146097 * era; // [0,146096]
103  uint16_t year_of_era = (day_of_era - day_of_era / 1460 + day_of_era / 36524
104  - day_of_era / 146096) / 365; // [0,399]
105  uint16_t year_prime = year_of_era + 400 * era; // [0,9999]
106  uint16_t day_of_year_prime = day_of_era
107  - (365 * year_of_era + year_of_era / 4 - year_of_era/100); // [0,365]
108  uint8_t month_prime = (5 * day_of_year_prime + 2) / 153;
109  uint16_t days_until_month_prime =
110  atc_convert_to_days_until_month_prime(month_prime);
111 
112  *day = day_of_year_prime - days_until_month_prime + 1; // [1,31]
113  *month = (month_prime < 10) ? month_prime + 3 : month_prime - 9; // [1,12]
114  *year = year_prime + ((*month <= 2) ? 1 : 0); // [1,9999]
115 }
