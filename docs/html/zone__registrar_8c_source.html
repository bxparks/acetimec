27 static uint16_t binary_search(
28  const AtcZoneInfo * const * registry,
29  uint16_t size,
30  uint32_t zone_id)
31 {
32  uint16_t a = 0;
33  uint16_t b = size;
34  while (1) {
35  uint16_t diff = b - a;
36  if (diff == 0) break;
37 
38  uint16_t c = a + diff / 2;
39  const AtcZoneInfo *curr = registry[c];
40  if (curr->zone_id == zone_id) return c;
41  if (zone_id < curr->zone_id) {
42  b = c;
43  } else {
44  a = c + 1;
45  }
46  }
47  return UINT16_MAX;
48 }
